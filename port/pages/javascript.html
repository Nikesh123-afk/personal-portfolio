    class TodoApp {
  constructor() {
    this.todos = JSON.parse(localStorage.getItem('todos')) || [];
    this.render();
  }
  
  addTodo(text) {
    const todo = {
      id: Date.now(),
      text: text,
      completed: false
    };
    this.todos.push(todo);
    this.save();
    this.render();
  }
  
  save() {
    localStorage.setItem('todos', JSON.stringify(this.todos));
  }
}

    Class Definition

    class Student {
    constructor(name, age, course) {
        this.name = name;
        this.age = age;
        this.course = course;
        this.grades = [];
    }
    
    addGrade(subject, grade) {
        this.grades.push({ subject, grade });
    }
    
    calculateGPA() {
        if (this.grades.length === 0) return 0;
        const total = this.grades.reduce(
            (sum, item) => sum + item.grade, 0
        );
        return (total / this.grades.length).toFixed(2);
    }
    
    getDetails() {
        return `${this.name}, Age: ${this.age}, Course: ${this.course}`;
    }
    
    // Static method
    static createHonorsStudent(name, age) {
        const student = new Student(name, age, 'Honors Program');
        return student;
    }
}

// Using the class
const student = new Student('John', 20, 'CET138');
student.addGrade('Math', 85);
student.addGrade('Science', 92);
console.log(student.calculateGPA()); // 88.50

    class Calculator {
    constructor() {
        this.display = document.getElementById('calculatorDisplay');
        this.currentInput = '0';
        this.shouldResetDisplay = false;
    }
    
    appendToDisplay(value) {
        if (this.shouldResetDisplay) {
            this.currentInput = '';
            this.shouldResetDisplay = false;
        }
        
        if (this.currentInput === '0' && value !== '.') {
            this.currentInput = value;
        } else {
            this.currentInput += value;
        }
        
        this.updateDisplay();
    }
    
    calculate() {
        try {
            // Safely evaluate the expression
            const result = Function('"use strict"; return (' + 
                this.currentInput.replace(/×/g, '*') + ')')();
            this.currentInput = result.toString();
            this.shouldResetDisplay = true;
            this.updateDisplay();
        } catch (error) {
            this.currentInput = 'Error';
            this.shouldResetDisplay = true;
            this.updateDisplay();
        }
    }
    
    clear() {
        this.currentInput = '0';
        this.updateDisplay();
    }
    
    updateDisplay() {
        this.display.value = this.currentInput;
    }
}

// Initialize calculator
const calc = new Calculator();
